## The Universal Input

All interactions start with natural language.

### Intent Detection

```python
class UniversalRouter:
    """Routes natural language to appropriate mode."""
    
    PATTERNS = {
        # Code
        ("build", "create app", "implement", "fix bug", "add feature"): "code",
        
        # Writing
        ("write chapter", "novel", "story", "screenplay", "blog"): "writing",
        
        # Documents
        ("draft report", "proposal", "memo", "letter", "resume"): "documents",
        
        # Planning
        ("plan", "organize", "todo", "schedule", "project"): "planning",
        
        # Notes
        ("note", "remember", "research", "learn about"): "notes",
        
        # Data
        ("analyze", "data", "spreadsheet", "chart", "numbers"): "data",
        
        # Communication
        ("email", "message", "write to", "respond"): "communication",
        
        # Design
        ("design", "mockup", "wireframe", "diagram"): "design",
    }
    
    async def route(self, goal: str, context: ProjectContext) -> Mode:
        # Use existing project context if available
        if context.current_mode:
            # Check if goal matches current mode
            if self.matches_mode(goal, context.current_mode):
                return context.current_mode
        
        # Detect from goal
        detected = await self.detect_mode(goal)
        
        # Confirm if ambiguous
        if detected.confidence < 0.8:
            return await self.clarify(goal, detected.options)
        
        return detected.mode
```

class AdaptiveLayout:

    async def compute_layout(
        self, 
        mode: Mode,

        context: ProjectContext,

        # Surface relevant panels

## Technical Architecture

### Unified Project Model

```python
@dataclass(frozen=True, slots=True)
class Project:
    """Universal project that spans all modes."""
    
    id: str
    name: str
    path: Path
    
    # Content by mode
    code_root: Path | None
    docs_root: Path | None
    notes_root: Path | None
    data_root: Path | None
    
    # Unified memory
    memory: Simulacrum
    
    # Cross-references
    links: list[CrossLink]
    
    # Automation
    workflows: list[Workflow]


@dataclass(frozen=True, slots=True)
class CrossLink:
    """Link between artifacts across modes."""
    
    source: ArtifactRef  # e.g., "notes/api-design.md:15"
    target: ArtifactRef  # e.g., "code/api/routes.py:45"
    link_type: LinkType  # IMPLEMENTS, REFERENCES, DEPENDS_ON, etc.
    confidence: float
```

### Unified Search

```python
class UnifiedSearch:
    """Search across all modes and content types."""
    
    async def search(
        self,
        query: str,
        filters: SearchFilters | None = None,
    ) -> list[SearchResult]:
        # Search all indexes in parallel
        results = await asyncio.gather(
            self.code_index.search(query),
            self.docs_index.search(query),
            self.notes_index.search(query),
            self.memory.search(query),
        )
        
        # Rank and dedupe
        return self.rank_results(results, query)
```
